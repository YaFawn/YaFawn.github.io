insta_df %>% 
  count(aisle) %>% 
  arrange(-n) %>% 
  mutate(aisle = fct_reorder(aisle, n, .desc = TRUE)) %>% 
  plot_ly(
    x = ~aisle, y = ~n, type = "bar", color = ~aisle, colors = "viridis"
  ) %>% 
  layout(title = "Items ordered in each aisle",
         xaxis  = list(title = "Aisle name"),
         yaxis  = list(title = "Items ordered")
         )
